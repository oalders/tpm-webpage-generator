package Apache::Hello;

use strict;
use Apache::Constants qw(:common);

sub handler {

    my $r = shift; # "response" object, inherited from the mod_perl framework

    $r->content_type('text/html');
    $r->send_http_header;

    my $host = $r->get_remote_host;

    my $counter;

    $counter++;

    $r->print(<<"END");
<HTML>

<HEAD>
<TITLE>
Hello There
</TITLE
</HEAD>

<BODY>
<H1>Hello, $host</H1>

<P>
Who would take my presentation seriously if its first example wasn't
a Hello World?

<P>
P.S.  This is mostly just lifted from the Eagle book.  But don't
tell anyone.

<P>
P.S.S.  The value of counter is now: $counter

</BODY>

</HTML>
END
  ;
    return OK;
}

1;  # gotta return true, until Perl 6... *sigh*
